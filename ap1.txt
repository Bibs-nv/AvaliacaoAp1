ls --> lista os arquivos e/ou pastas dentro da pasta
git add . --> adiciona os arquivos para a "sacola"
git commit -m "Enviando o primeiro commit ap1" --> fecha sacola 
e coloca etiqueta
git push --set-upstream origin master --> envia para o repositório
